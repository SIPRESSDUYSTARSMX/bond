48 explicit basic_thread_pool(unsigned int numThreads)
49 {
50 if (numThreads == 0)
51 {
52 throw InvalidThreadCount{};
53 }
54
55 auto& service = *_service;
56 for (unsigned int i = 0; i < numThreads; ++i)
57 {
58 service.threads.emplace_back(
59 [&service]
60 {
61 service.run();
62 });
63 }
64 }
