58 inline ::grpc::ByteBuffer to_byte_buffer(const OutputBuffer& output)
59 {
60 struct Buffers
61 : boost::intrusive_ref_counter<Buffers>,
62 boost::container::small_vector<blob, 8>
63 {};
64
65 boost::intrusive_ptr<Buffers> buffers{ new Buffers };
66 output.GetBuffers(*buffers);
67
68 boost::container::small_vector<::grpc::Slice, 8> slices;
69 slices.reserve(buffers->size());
70
71 for (blob& data : *buffers)
72 {
73 data = blob_prolong(std::move(data));
74
75 slices.emplace_back(
76 const_cast<void*>(data.data()), // The buffer is not expected to be modified, but
77 // we have to const_cast because ::grpc::Slice ctor
78 // only takes void*.
79 data.size(),
80 [](void* arg) { intrusive_ptr_release(static_cast<Buffers*>(arg)); },
81 buffers.get());
82
83 intrusive_ptr_add_ref(buffers.get());
84 }
85
86 return ::grpc::ByteBuffer{ slices.data(), slices.size() };
87 }

100 inline InputBuffer from_byte_buffer(const ::grpc::ByteBuffer& buffer)
101 {
102 std::vector<::grpc::Slice> slices;
103
104 auto status = buffer.Dump(&slices);
105 if (!status.ok())
106 {
107 throw GrpcException{ status };
108 }
109
110 const auto length = buffer.Length();
111 auto buff = boost::make_shared_noinit<char[]>(length);
112
113 char* dest = buff.get();
114 for (auto& s : slices)
115 {
116 std::memcpy(dest, s.begin(), s.size());
117 dest += s.size();
118 }
119
120 // TODO: create a Bond input stream over ::grpc::ByteBuffer to avoid
121 // having to make this copy into a blob.
122 blob data{ buff, static_cast<uint32_t>(length) };
123 return InputBuffer{ data };
124 }
