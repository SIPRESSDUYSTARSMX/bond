93 static server Start(::grpc::ServerBuilder& builder, service_collection services)
94 {
95 auto cq = builder.AddCompletionQueue();
96
97 for (const auto& item : boost::combine(services.services(), services.names()))
98 {
99 auto& service = item.get<0>();
100
101 if (const auto& host = item.get<1>())
102 {
103 builder.RegisterService(host.value(), service->grpc_service());
104 }
105 else
106 {
107 builder.RegisterService(service->grpc_service());
108 }
109
110 service->SetCompletionQueue(cq.get());
111 }
112
113 if (auto svr = builder.BuildAndStart())
114 {
115 return server{
116 std::move(svr),
117 std::move(services.services()),
118 std::unique_ptr<io_manager>{ new io_manager{
119 std::thread::hardware_concurrency(), /*delay=*/ false, std::move(cq) } } };
120 }
121
122 throw ServerBuildException{};
123 }
